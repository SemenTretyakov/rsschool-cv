const bracketSequence = function(str) {
    let count = 0;
    for (let i = 0; i < str.length; i += 1) {
        if (str[i] === '(') {
            count = count + 1;
        }   
        else if (str[i] === ')') {
            count = count - 1;
        }   
    }
    if (count === 0) {
        alert('Баланс скобок соблюдается');
    } else if (count === -1) {
        alert('Нашли «некомпенсируемую» закрывающую скобку');
    } else if (count > 0) {
        alert('Баланс скобок не соблюдается');
    }
}
